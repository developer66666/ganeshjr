2. Case Manipulation

  • Upper( )
  • Lower( )
  • INITCAP( )
SELECT first_name, UPPER(first_name), LOWER(first_name), INITCAP(first_name) FROM employees;
SELECT INITCAP('welcome to oracle session') FROM dual;
SELECT INITCAP('welcoMe to oracLe sEssion') FROM dual;        -- Even the center word is caps it change.
SELECT employee_id, first_name, hire_date, salary, department_id
  FROM employees
WHERE LOWER(first_name) = LOWER('Steven');

3. Character Manipulation

  • LENGTH( )
  • CONCAT( )
  • REVERSE( )
  • SUBSTR( )
  • INSTR( )
  • LPAD( ) | RPAD( )
  • LTRIM( ) | RTRIM( ) | TRIM( )
  • REPLACE( ) | TRANSLATE( )
SELECT first_name
     , LENGTH(first_name) length
     , CONCAT(first_name, last_name) "Name"
     , INITCAP(REVERSE(first_name)) "REV_NAME"     --255  --nested function we can write upto 255 function
  FROM employees;

Find Total No of 'a' in Ganeshbabu

SELECT REGEXP_COUNT ('Ganeshbabu', 'a') FROM dual;  

SELECT 'Ganeshbabu' name, LENGTH('Ganeshbabu') - LENGTH(REPLACE('Ganeshbabu','a','')) length FROM dual;

SUBSTR( )

SELECT first_name
     , SUBSTR(first_name, 1, 3) "SUB_01"
     , SUBSTR(first_name, 2, 3) "SUB_02"
     , SUBSTR(first_name, 2   ) "SUB_03"
     , SUBSTR(first_name,-3, 3) "SUB_04"
     , SUBSTR(first_name,-2, 2) "SUB_05"
     , SUBSTR(first_name,-8   ) "SUB_06"
     , SUBSTR(first_name,-8, 3) "SUB_07"
     , SUBSTR(first_name, 8, 3) "SUB_08"
  FROM employees;

INSTR( )

SELECT first_name
     , INSTR(LOWER(first_name), LOWER('A')) "INST_01"
     , INSTR(LOWER(first_name), 'a')        "INST_02"
     , INSTR(LOWER(first_name), 'a', 1, 2)  "INST_03"        
     , INSTR(LOWER(first_name), 'a', 1, 3)  "INST_04"
  FROM employees;

SELECT
   INSTR('Ganesh Babu J R', 'e')  "FIRST"
 , INSTR('Ganesh Babu J R', 'a', 1, 2) "SECOND"
 FROM dual;

Display all the first name which contain second character as a by using INSTR( )

LPAD() and RPAD()

SELECT first_name, salary
       , LPAD(salary, 8, 0)    "LPAD"
       , RPAD(salary, 8, 'X')    "RPAD"
    FROM employees;

LTRIM(), RTRIM(), TRIM()

SELECT '00000000098760870000'        "DATE"
     , LTRIM('00000000098760870000','0') "LTRIM"
     , RTRIM('00000000098760870000','0') "RTRIM"
     , LTRIM(RTRIM('00000000098760870000','0'),'0') "TRIM"
     , TRIM('0' FROM '00000000098760870000')        "TRIM"
  FROM dual;

REPLACE() and TRANSLATE()

SELECT 'Welcome to Oracle Session'
     , REPLACE('Welcome to Oracle Session', 'lc', 'XY') "Replace"
     , TRANSLATE('Welcome to Oracle Session', 'lc', 'XY') "Translate"
FROM dual;

Date Functions

MONTHS_BETWEEN

SELECT MONTHS_BETWEEN(SYSDATE,TO_DATE('13-JUN-22')) FROM dual;

SELECT NEXT_DAY(SYSDATE,'SUNDAY') FROM dual;

General (NULL) Functions

NVL()

SELECT employee_id, first_name, hire_date, salary, commission_pct, NVL(commission_pct,0) FROM employees;

NVL2()

SELECT employee_id, first_name, hire_date, salary, commission_pct, NVL2(commission_pct,1,0) FROM employees;

NULLIF()

SELECT NULLIF(1,2), NULLIF(1,1), NULLIF(2,2), NULLIF(2,1) FROM dual;

COALESCE()

SELECT COALESCE(NULL,NULL,78,NULL,NULL, 68) FROM dual;

Group Functions

Basic Aggregate Functions

SELECT MIN(salary) FROM employees;
SELECT MAX(salary) FROM employees;
SELECT SUM(salary) FROM employees;
SELECT COUNT(salary) FROM employees;
SELECT ROUND(AVG(salary)) FROM employees;

GROUP BY Examples

SELECT department_id, MAX(salary) FROM employees
GROUP BY department_id
ORDER BY 1;

SELECT department_id, COUNT(*)
FROM employees
GROUP BY department_id
ORDER BY 1;

HAVING Clause

SELECT department_id, COUNT(*)
FROM employees
GROUP BY department_id
HAVING COUNT(*) >=30
ORDER BY 1;

Practice Queries

Display all the employees hired in 1994

SELECT employee_id, first_name, TO_CHAR(hire_date,'YYYY') FROM employees
WHERE TO_CHAR(hire_date,'YYYY') IN '1994';

Display all the employees hired in week days

SELECT employee_id, first_name, TO_CHAR(hire_date,'Day') FROM employees
WHERE TRIM(TO_CHAR(hire_date,'Day')) IN ('Monday','Tuesday','Wednesday','Thursday','Friday');

No of Employees Reporting to Each Manager

SELECT manager_id, COUNT(*) 
FROM employees
WHERE manager_id IS NOT NULL
GROUP BY manager_id;

Find Duplicate Records

SELECT first_name,COUNT(*)
FROM employees
GROUP BY first_name
HAVING COUNT(*)>1;
